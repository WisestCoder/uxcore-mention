/**
 * [parseStrByDelimiter description]
 * @method parseStrByDelimiter
 * @param  {[string]} str       = ''  [origin str]
 * @param  {[string]} delimiter = '@' [delimiter str]
 * @return {[string]}
 */
function parseStrByDelimiter() {
  var str = arguments.length <= 0 || arguments[0] === undefined ? '' : arguments[0];
  var delimiter = arguments.length <= 1 || arguments[1] === undefined ? '@' : arguments[1];
 
  var idx = str.lastIndexOf(delimiter);
  var ret = void 0;
  if (idx !== -1) {
    ret = str.substring(idx + 1);
  } else {
    ret = '';
  }
  return ret;
}
 
/**
 * [get window scroll offset]
 * @method getScrollOffset
 */
function getScrollOffset() {
  var offset = {};
  if (window.pageXOffset) {
    offset.x = window.pageXOffset;
  } else {
    offset.x = document.documentElement.scrollLeft;
  }
  if (window.pageYOffset) {
    offset.y = window.pageYOffset;
  } else {
    offset.y = document.documentElement.scrollTop;
  }
  return offset;
}
 
function getCaretOffset(element) {
  var start = void 0;
  var end = void 0;
  if (element.setSelectionRange) {
    start = element.selectionStart;
    end = element.selectionEnd;
  } else {
    element.focus();
    var range = document.selection.createRange();
    if (element.tagName.toUpperCase() === 'TEXTAREA') {
      // textarea
      var duplicate = range.duplicate();
      duplicate.moveToElementText(element);
      start = -1;
      while (duplicate.inRange(range)) {
        duplicate.moveStart('character');
        start += 1;
      }
    } else if (element.tagName.toUpperCase() === 'INPUT') {
      // input
      range.moveStart('character', -element.value.length);
      start = range.text.length;
    }
    end = start + range.text.length;
  }
  return {
    start: start,
    end: end
  };
}
 
function getElementOffset(element) {
  var box = element.getBoundingClientRect();
  var doc = element.ownerDocument;
  var body = doc.body;
  var clientTop = doc.documentElement.clientTop || body.clientTop || 0;
  var clientLeft = doc.documentElement.clientLeft || body.clientLeft || 0;
  var top = box.top + (window.pageYOffset || doc.documentElement.scrollTop) - clientTop;
  var left = box.left + (window.pageXOffset || doc.documentElement.scrollLeft) - clientLeft;
 
  return {
    left: left,
    top: top
  };
}
